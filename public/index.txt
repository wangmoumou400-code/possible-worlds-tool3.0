        <p>基于可能世界理论的AI创意写作助手</p>

    <div class="content">
        <textarea id="userInput" rows="5" placeholder="请输入你的故事片段或大纲..."></textarea>
        <button id="generateBtn">✨ 探索可能世界</button>
        <div id="resultArea"></div>
        <div class="info-note">💡 基于Ryan (1991) 九种可及性关系生成反事实变体。请选择3-5条最有启发性的方向融入故事。</div>
    </div>
    <div class="footer">🎓 研究用途 | 基于可能世界理论设计</div>

<script>
    const generateBtn = document.getElementById('generateBtn');
    const userInput = document.getElementById('userInput');
    const resultArea = document.getElementById('resultArea');

    let currentVariations = [];      // 存储当前9条变体
    let selectedVariationsArray = []; // 存储最终选择的变体

    // 生成并下载JSON记录文件
    function downloadExperimentData() {
        const record = {
            timestamp: new Date().toISOString(),
            userInput: userInput.value.trim(),
            allVariations: currentVariations,
            selectedVariations: selectedVariationsArray,
            selectionCount: selectedVariationsArray.length
        };
        const jsonStr = JSON.stringify(record, null, 2);
        const blob = new Blob([jsonStr], { type: 'application/json' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `experiment_data_${Date.now()}.json`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
        alert('实验数据已保存为JSON文件，请妥善保管！');
    }

    // 渲染变体列表（带点击选择）
    function renderVariations(variations) {
        let html = '<div style="margin-top:24px;"><h3>🌟 你可能感兴趣的可能世界</h3>';
        html += '<div class="selection-info">请选择 <span class="selected-count" id="selectedCount">0</span> / 3-5 条最有启发性的变体（点击卡片切换选择）</div>';
        html += '<div id="variationsList">';
        variations.forEach((v, idx) => {
            html += `
                <div class="variation-card" data-idx="${idx}">
                    <div class="variation-type">${v.type}</div>
                    <div class="variation-text">“${v.text}”</div>
                </div>
            `;
        });
        html += '</div>';
        html += '<div style="margin-top:20px; display:flex; gap:12px;">';
        html += '<button id="confirmBtn" class="btn-primary">✓ 确认选择并开始写作</button>';
        html += '<button id="resetSelectionBtn" class="btn-secondary">⟳ 清空选择</button>';
        html += '<button id="exportDataBtn" class="btn-export">📥 导出实验数据</button>';
        html += '</div></div>';
        resultArea.innerHTML = html;

        const cards = document.querySelectorAll('.variation-card');
        const selectedSet = new Set();  // 存储选中索引
        const selectedCountSpan = document.getElementById('selectedCount');

        function updateSelectedCount() {
            selectedCountSpan.textContent = selectedSet.size;
            if (selectedSet.size < 3) {
                selectedCountSpan.style.color = '#dc3545';
            } else if (selectedSet.size > 5) {
                selectedCountSpan.style.color = '#ffc107';
            } else {
                selectedCountSpan.style.color = '#198754';
            }
        }

        cards.forEach(card => {
            card.addEventListener('click', () => {
                const idx = parseInt(card.dataset.idx);
                if (selectedSet.has(idx)) {
                    selectedSet.delete(idx);
                    card.classList.remove('selected');
                } else {
                    selectedSet.add(idx);
                    card.classList.add('selected');
                }
                updateSelectedCount();
            });
        });

        // 确认选择
        document.getElementById('confirmBtn').addEventListener('click', () => {
            const selected = Array.from(selectedSet).map(i => variations[i]);
            if (selectedSet.size < 3) {
                if (confirm(`您只选择了 ${selectedSet.size} 条变体。建议选择3-5条以获得更丰富的灵感。是否继续？`)) {
                    finalizeSelection(selected);
                }
            } else if (selectedSet.size > 5) {
                if (confirm(`您选择了 ${selectedSet.size} 条变体。建议不超过5条以免分散注意力。是否继续？`)) {
                    finalizeSelection(selected);
                }
            } else {
                finalizeSelection(selected);
            }
        });

        // 清空选择
        document.getElementById('resetSelectionBtn').addEventListener('click', () => {
            selectedSet.clear();
            cards.forEach(card => card.classList.remove('selected'));
            updateSelectedCount();
        });

        // 导出数据（无需先确认）
        document.getElementById('exportDataBtn').addEventListener('click', () => {
            if (currentVariations.length === 0) {
                alert('尚未生成任何变体，无法导出。');
                return;
            }
            downloadExperimentData();
        });

        function finalizeSelection(selected) {
            selectedVariationsArray = selected;
            // 显示确认信息，并提供开始写作按钮
            let summaryHtml = '<div class="selection-info" style="margin-top:16px;">✅ 已选择 ' + selected.length + ' 条变体。你可以开始写作了！</div>';
            summaryHtml += '<button id="startWritingBtn" class="btn-primary">✍️ 开始写作</button>';
            resultArea.insertAdjacentHTML('beforeend', summaryHtml);
            document.getElementById('startWritingBtn').addEventListener('click', () => {
                alert('请返回写作界面，开始正式写作。');
                // 可选：此处可自动清除选择区域，但保留界面让用户继续操作
            });
            // 全局保存以备导出
            window.selectedVariations = selected;
        }
    }

    generateBtn.addEventListener('click', async () => {
        const text = userInput.value.trim();
        if (!text) {
            resultArea.innerHTML = '<div style="color:red;">请先输入故事片段或大纲</div>';
            return;
        }
        resultArea.innerHTML = '<div class="loading">✨ 正在探索平行宇宙...</div>';
        try {
            const res = await fetch('/api/generate', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ userInput: text })
            });
            const data = await res.json();
            if (data.error) throw new Error(data.error);
            currentVariations = data.variations;
            renderVariations(currentVariations);
        } catch (err) {
            resultArea.innerHTML = '<div style="color:red;">生成失败，请稍后重试</div>';
        }
    });
</script>
